// Class representing a person with name, gender, and age attributes
public class Person {
    // Attributes of the Person class
    String name;  // Name of the person
    String gender;  // Gender of the person
    int age;  // Age of the person

    // Method to display the details of the person
    void show() {
        // Printing the name of the person
        System.out.println("Name : " + name);
        // Printing the gender of the person
        System.out.println("Gender : " + gender);
        // Printing the age of the person
        System.out.println("Age : " + age);
    }
}
